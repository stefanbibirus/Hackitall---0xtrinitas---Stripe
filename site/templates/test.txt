    <script>
        const searchBox = document.getElementById('searchBox');
        const searchIcon = document.getElementById('searchIcon');
        const searchContainer = document.getElementById('searchContainer');
        const productsContainer = document.getElementById('productsContainer');
        const productsGrid = document.getElementById('productsGrid');
        const filtersBtn = document.getElementById('filtersBtn');
        const filterInputContainer = document.getElementById('filterInputContainer');
        const filterInput = document.getElementById('filterInput');
        const filterBubbles = document.getElementById('filterBubbles');
        const floatingPlus = document.getElementById('floatingPlus');
        const welcomeText = document.querySelector('h1');

        let filters = [];
        let isSearchActive = false;

        // JSON data with welcome messages
        const welcomeMessages = [
            "Welcome back!",
            "Hello again!",
            "Great to see you!",
            "Ready to explore?",
            "Let's find something!",
            "What are you looking for?",
            "Happy shopping!",
            "Discover something new!",
            "Time to browse!",
            "Welcome aboard!"
        ];

        // Function to pick a random welcome message
        function setRandomWelcomeMessage() {
            const randomIndex = Math.floor(Math.random() * welcomeMessages.length);
            welcomeText.textContent = welcomeMessages[randomIndex];
        }

        // Set random message on page load
        setRandomWelcomeMessage();

        // Sample product data
        const sampleProducts = [
            { title: 'Premium Headphones', price: '$99.99', description: 'High-quality wireless headphones with noise cancellation' },
            { title: 'Smart Watch', price: '$249.99', description: 'Feature-packed smartwatch with health tracking' },
            { title: 'Laptop Stand', price: '$49.99', description: 'Ergonomic aluminum laptop stand' },
            { title: 'Wireless Mouse', price: '$29.99', description: 'Comfortable wireless mouse with precision tracking' },
            { title: 'Mechanical Keyboard', price: '$129.99', description: 'RGB mechanical keyboard with tactile switches' },
            { title: 'USB-C Hub', price: '$39.99', description: 'Multi-port USB-C hub with HDMI and card reader' },
            { title: 'Webcam 4K', price: '$89.99', description: 'Ultra HD webcam with auto-focus' },
            { title: 'Phone Case', price: '$19.99', description: 'Protective case with premium materials' },
            { title: 'Portable Charger', price: '$34.99', description: '20000mAh power bank with fast charging' }
        ];

        // Show filter input when Filters button is clicked
        filtersBtn.addEventListener('click', () => {
            filtersBtn.classList.add('hidden');
            filterInputContainer.classList.add('show');
            filterInput.focus();
        });

        // Add filter bubble when Enter is pressed in filter input
        filterInput.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                const filterText = filterInput.value.trim();
                if (filterText) {
                    addFilterBubble(filterText);
                    filterInput.value = '';
                }
            }
        });

        // Function to add filter bubble
        function addFilterBubble(text) {
            filters.push(text);
            
            const bubble = document.createElement('div');
            bubble.className = 'filter-bubble';
            bubble.innerHTML = `
                ${text}
                <button class="remove-btn" onclick="removeFilterBubble(this, '${text}')">Ã—</button>
            `;
            filterBubbles.appendChild(bubble);
        }

        // Function to remove filter bubble
        window.removeFilterBubble = function(btn, text) {
            const bubble = btn.parentElement;
            bubble.style.animation = 'bubbleOut 0.3s ease';
            
            setTimeout(() => {
                bubble.remove();
                filters = filters.filter(f => f !== text);
            }, 300);
        };

        function performSearch() {
            const searchTerm = searchBox.value.trim();
            
            if (searchTerm) {
                isSearchActive = true;
                
                // Hide the Filters button when searching
                filtersBtn.classList.add('hidden');
                filterInputContainer.classList.remove('show');
                
                // Move search to top
                searchContainer.classList.add('top');
                
                // Show products and floating plus button
                productsContainer.classList.add('show');
                floatingPlus.classList.add('show');
                
                // Clear existing products
                productsGrid.innerHTML = '';
                
                // Add products
                sampleProducts.forEach((product, index) => {
                    const productCard = document.createElement('div');
                    productCard.className = 'product-card';
                    productCard.innerHTML = `
                        <div class="product-image">Product Image</div>
                        <div class="product-title">${product.title}</div>
                        <div class="product-price">${product.price}</div>
                        <div class="product-description">${product.description}</div>
                    `;
                    productsGrid.appendChild(productCard);
                    
                    // Add staggered animation
                    setTimeout(() => {
                        productCard.style.animation = 'fadeIn 0.5s ease forwards';
                    }, index * 50);
                });
            }
        }

        // Floating plus button - reset to search page
        floatingPlus.addEventListener('click', () => {
            isSearchActive = false;
            
            // Reset search container
            searchContainer.classList.remove('top');
            
            // Hide products and floating button
            productsContainer.classList.remove('show');
            floatingPlus.classList.remove('show');
            
            // Show filters button again
            filtersBtn.classList.remove('hidden');
            filterInputContainer.classList.remove('show');
            
            // Clear search box
            searchBox.value = '';
            
            // Clear filter bubbles
            filterBubbles.innerHTML = '';
            filters = [];
        });

        // Search on Enter key
        searchBox.addEventListener('keypress', (e) => {
            if (e.key === 'Enter') {
                performSearch();
            }
        });

        // Search on icon click
        searchIcon.addEventListener('click', performSearch);

        // Add animations
        const style = document.createElement('style');
        style.textContent = `
            @keyframes fadeIn {
                from {
                    opacity: 0;
                    transform: translateY(20px);
                }
                to {
                    opacity: 1;
                    transform: translateY(0);
                }
            }
            
            @keyframes bubbleOut {
                from {
                    opacity: 1;
                    transform: scale(1);
                }
                to {
                    opacity: 0;
                    transform: scale(0.8);
                }
            }
        `;
        document.head.appendChild(style);
    </script>
